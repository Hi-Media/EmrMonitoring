 23: /**
 24:  * Getopt.PHP allows for easy processing of command-line arguments.
 25:  * It is a more powerful, object-oriented alternative to PHP's built-in getopt() function.
 26:  *
 27:  * @version 1.0
 28:  * @link    https://github.com/ulrichsg/getopt-php
 29:  */

 32:     const NO_ARGUMENT = 0;
 33:     const REQUIRED_ARGUMENT = 1;
 34:     const OPTIONAL_ARGUMENT = 2;

 46:      * Create a new Getopt object.
 47:      *
 48:      * The argument $options can be either a string in the format accepted by the PHP library
 49:      * function getopt() or an array
 50:      *
 51:      * @param mixed $options Array of options, a String, or null
 52:      *
 53:      * @link https://www.gnu.org/s/hello/manual/libc/Getopt.html GNU Getopt manual
 54:      */
 55:     public function __construct($options = null) {
 56:         if ($options !== null) {

 61:     /**
 62:      * Evaluate the given arguments. These can be passed either as a string or as an array.
 63:      * If nothing is passed, the running script's command line arguments are used.
 64:      *
 65:      * A {@link \UnexpectedValueException} or {@link \InvalidArgumentException} is thrown
 66:      * when the arguments are not well-formed or do not conform to the options passed by the user.
 67:      *
 68:      * @param mixed $arguments optional ARGV array or space separated string
 69:      *
 70:      * @return void
 71:      */
 72:     public function parse($arguments = null) {
 73:         $this->options = array();
 74:         if (!isset($arguments)) {
 75:             global $argv;
 76:             $arguments = $argv;
 77:             $this->scriptName = array_shift($arguments); // $argv[0] is the script's name
 78:         } elseif (is_string($arguments)) {
 79:             $this->scriptName = $_SERVER['PHP_SELF'];
 80:             $arguments = explode(' ', $arguments);
 81:         }
 82: 
 83:         $num_args = count($arguments);
 84:         for ($i = 0; $i < $num_args; ++$i) {
 85:             $arg = trim($arguments[$i]);
 86:             if (empty($arg)) {
 87:                 continue;
 88:             }
 89:             if ($arg == '--' || mb_substr($arg, 0, 1) != '-') {
 90:                 // no more options, treat the remaining arguments as operands
 91:                 $first_operand_index = $arg == '--' ? $i + 1 : $i;
 92:                 $this->operands = array_slice($arguments, $first_operand_index);
 93:                 break;
 94:             }
 95:             if (mb_substr($arg, 0, 2) == '--') {
 96:                 // long option
 97:                 $option = mb_substr($arg, 2);
 98:                 if (strpos($option, '=') === false) {
 99:                     if ($i < $num_args - 1
100:                             && mb_substr($arguments[$i + 1], 0, 1) != '-'
101:                             && $this->optionHasArgument($option, true)) {
102:                         $value = $arguments[$i + 1];
103:                         ++$i;
104:                     } else {
105:                         $value = null;
106:                     }
107:                 } else {
108:                     list($option, $value) = explode('=', $option, 2);
109:                 }
110:                 $this->addOption($option, $value, true);
111:             } else {
112:                 // short option
113:                 $option = mb_substr($arg, 1);
114:                 if (mb_strlen($option) > 1) {
115:                     // multiple options strung together
116:                     foreach ($this->mb_str_split($option, 1) as $ch) {
117:                         $this->addOption($ch, null, false);
118:                     }
119:                 } else {
120:                     if ($i < $num_args - 1
121:                             && mb_substr($arguments[$i + 1], 0, 1) != '-'
122:                             && $this->optionHasArgument($option, false)) {
123:                         $value = $arguments[$i + 1];
124:                         ++$i;
125:                     } else {
126:                         $value = null;
127:                     }
128:                     $this->addOption($option, $value, false);
129:                 }
130:             }
131:         } // endfor
132: 
133:         // remove '--' from operands array
134:         $operands = array();
135:         foreach($this->operands as $operand) {
136:             if ($operand !== '--') {
137:                 $operands[] = $operand;
138:             }
139:         }
140:         $this->operands = $operands;
141:     }

143:     /**
144:      * Return the value of the given option. Must be invoked after parse().
145:      *
146:      * The return value can be any of the following:
147:      * <ul>
148:      *   <li><b>null</b> if the option is not given</li>
149:      *   <li><b>an integer</b> if the option is given without argument. The
150:      *       returned value is the number of occurrences of the option.</li>
151:      *   <li><b>a string</b> if the option is given with an argument. The
152:      *       returned value is that argument.</li>
153:      * </ul>
154:      *
155:      * @param string $name The (short or long) option.
156:      *
157:      * @return mixed
158:      */

183:     /**
184:      * @param int $padding Number of characters to pad output of options to
185:      *
186:      * @return string help message for given options.
187:      */
188:     public function getHelpText($padding = 25) {
189:         $help_text = sprintf("Usage: %s [options] [operands]\n", $this->scriptName);
190:         $help_text .= "Options:\n";
191:         foreach ($this->optionList as $name => $option) {
192:             list($short, $long, $arg, $description) = $option;
193:             switch ($arg) {
194:                 case self::NO_ARGUMENT: $arg = ''; break;
195:                 case self::REQUIRED_ARGUMENT: $arg = "<arg>"; break;
196:                 case self::OPTIONAL_ARGUMENT: $arg = "[<arg>]"; break;
197:             }
198:             $short = ($short) ? '-'.$short : '';
199:             $long = ($long) ? '--'.$long : '';
200:             if ($short && $long) {
201:                 $options = $short.', '.$long;
202:             } else if ($short) {
203:                 $options = $short;
204:             } else {
205:                 $options = $long;
206:             }
207:             $padded = str_pad(sprintf("  %s %s", $options, $arg), $padding);
208:             $help_text .= sprintf("%s %s\n", $padded, $description);
209:         }
210:         return $help_text;
211:     }

223:     /**
224:      * Parse an option string.
225:      *
226:      * @param string $string the option string
227:      *
228:      * @throws \InvalidArgumentException
229:      * @return array
230:      * @internal
231:      */
232:     protected function parseOptionString($string) {
233:         if (!mb_strlen($string)) {
234:             throw new \InvalidArgumentException('Option string must not be empty');
235:         }
236:         $option_list = array();
237:         $eol = mb_strlen($string) - 1;
238:         $next_can_be_colon = false;
239:         for ($i = 0; $i <= $eol; ++$i) {
240:             $ch = $string[$i];
241:             if (!preg_match('/^[A-Za-z]$/', $ch)) {
242:                 $colon = $next_can_be_colon ? " or ':'" : '';
243:                 throw new \InvalidArgumentException("Option string is not well formed: "
244:                         . "expected a letter$colon, found '$ch' at position " . ($i + 1));
245:             }
246:             if ($i == $eol || $string[$i + 1] != ':') {
247:                 $option_list[] = array($ch, null, self::NO_ARGUMENT);
248:                 $next_can_be_colon = true;
249:             } elseif ($i < $eol - 1 && $string[$i + 2] == ':') {
250:                 $option_list[] = array($ch, null, self::OPTIONAL_ARGUMENT);
251:                 $i += 2;
252:                 $next_can_be_colon = false;
253:             } else {
254:                 $option_list[] = array($ch, null, self::REQUIRED_ARGUMENT);
255:                 ++$i;
256:                 $next_can_be_colon = true;
257:             }
258:         }
259:         return $option_list;
260:     }

262:     /**
263:      * Check that the argument conforms to Getopt.PHP's option array rules.
264:      * Throws an exception on failure.
265:      *
266:      * @param array $options the option list
267:      *
268:      * @throws \InvalidArgumentException
269:      * @return array the validated options array
270:      * @internal
271:      */
272:     protected function validateOptions(array $options) {
273:         $valid_argument_specs = array(
274:             self::NO_ARGUMENT, self::OPTIONAL_ARGUMENT, self::REQUIRED_ARGUMENT
275:         );
276:         if (empty($options)) {
277:             throw new \InvalidArgumentException('No options given');
278:         }
279:         foreach ($options as &$option) {
280:             if (!is_array($option) || count($option) < 3) {
281:                 throw new \InvalidArgumentException("Too few fields in argument, must be 3 (short/long/type)");
282:             }
283:             if (!(is_null($option[0]) || preg_match("/^[a-zA-Z]$/", $option[0]))) {
284:                 throw new \InvalidArgumentException("First component of option must be "
285:                         . "null or a letter, found '" . $option[0] . "'");
286:             }
287:             if (!(is_null($option[1]) || preg_match("/^[a-zA-Z0-9_-]*$/", $option[1]))) {
288:                 throw new \InvalidArgumentException("Second component of option must be "
289:                         . "null or an alphanumeric string, found '" . $option[1] . "'");
290:             }
291:             if (!mb_strlen($option[0]) && !mb_strlen($option[1])) {
292:                 throw new \InvalidArgumentException("The short and long name of an option must not both be empty");
293:             }
294:             if (!in_array($option[2], $valid_argument_specs, true)) {
295:                 throw new \InvalidArgumentException("Third component of option must be one of "
296:                         . "Getopt::NO_ARGUMENT, Getopt::OPTIONAL_ARGUMENT and Getopt::REQUIRED_ARGUMENT");
297:             }
298:             if (!isset($option[3])) {
299:                 $option[3] = ""; // description
300:             }
301:         }
302:         return $options;
303:     }

305:     /**
306:      * Add an option to the list of known options.
307:      *
308:      * @param string $option the option's name
309:      * @param string $value the option's value (or null)
310:      * @param boolean $is_long whether the option name is long or short
311:      *
312:      * @throws \UnexpectedValueException
313:      * @return void
314:      * @internal
315:      */
316:     protected function addOption($option, $value, $is_long) {
317:         foreach ($this->optionList as $opt) {
318:             if (($is_long && $opt[1] == $option) || (!$is_long && $opt[0] == $option)) {
319:                 if ($opt[2] == self::REQUIRED_ARGUMENT && !mb_strlen($value)) {
320:                     throw new \UnexpectedValueException("Option '$option' must have a value");
321:                 }
322:                 // for no-argument options, check if they are duplicate
323:                 if ($opt[2] == self::NO_ARGUMENT) {
324:                     $old_value = $this->getOption($option);
325:                     $value = is_null($old_value) ? 1 : $old_value + 1;
326:                 }
327:                 // for optional-argument options, set value to 1 if none was given
328:                 if (!mb_strlen($value)) {
329:                     $value = 1;
330:                 }
331:                 // add both long and short names (if they exist) to the option array to facilitate lookup
332:                 if (mb_strlen($opt[0]) > 0) {
333:                     $this->options[$opt[0]] = $value;
334:                 }
335:                 if (mb_strlen($opt[1]) > 0) {
336:                     $this->options[$opt[1]] = $value;
337:                 }
338:                 return;
339:             }
340:         }
341:         throw new \UnexpectedValueException("Option '$option' is unknown");
342:     }

344:     /**
345:      * Return true if the given option can take an argument, false if it can't or is unknown.
346:      *
347:      * @param string $name the option's name
348:      * @param boolean $is_long whether it is a long option
349:      *
350:      * @return boolean
351:      * @internal
352:      */
353:     protected function optionHasArgument($name, $is_long) {
354:         foreach ($this->optionList as $option) {
355:             if ((!$is_long && $option[0] == $name)
356:                     || ($is_long && $option[1] == $name)) {
357:                 return $option[2] != self::NO_ARGUMENT;
358:             }
359:         }
360:         return false;
361:     }

376:      * Parses and Adds options
377:      * The argument $options can be either a string in the format accepted by the PHP library
378:      * function getopt() or an array
379:      *
380:      * @param mixed $options Array of options, a String, or null
381:      * @throws \InvalidArgumentException
382:      * @return array
383:      */
384:     public function addOptions($options)
385:     {
386:         if (is_string($options)) {
387:             return $this->addParsedOptions($this->parseOptionString($options));
388:         }
389: 
390:         if (is_array($options)) {
391:             return $this->addParsedOptions($this->validateOptions($options));
392:         }
393: 
394:         throw new \InvalidArgumentException("Getopt(): argument must be string or array");
395:     }
